\newcommand{\hstrasse}[4]{%
    \begin{pgfonlayer}{background}
      \draw[randstn] ({#1-2},{#2-2}) rectangle ({#3+1},{#2+1});
    \end{pgfonlayer}{background}
    \draw[strasse] (#1-1,#2-1) rectangle (#3,#2);
}
\newcommand{\vstrasse}[4]{%
    \begin{pgfonlayer}{background}
      \draw[randstn] ({#1-2},{#2-2}) rectangle ({#1+1},{#3+1});
    \end{pgfonlayer}{background}
    \draw[strasse] (#1-1,#2-1) rectangle (#1,#3);
}

\newcommand{\kreuzung}[5]{%
    \draw[kreuz]  ({#1-1},{#2-1}) rectangle ({#1+0},{#2+0});
    \draw[#3]        ({#1-2},{#2-1}) rectangle ({#1-1},{#2-0});
    \draw[#3]        ({#1+0},{#2-1}) rectangle ({#1+1},{#2+0});
    \draw[#3]        ({#1-1},{#2-4}) rectangle ({#1+0},{#2-3});
    \draw[#3]        ({#1-1},{#2+2}) rectangle ({#1+0},{#2+3});
    \draw[#4]        ({#1-3},{#2-1}) rectangle ({#1-2},{#2+0});
    \draw[#4]        ({#1+1},{#2-1}) rectangle ({#1+2},{#2+0});
    \draw[#4]        ({#1-1},{#2-3}) rectangle ({#1+0},{#2-2});
    \draw[#4]        ({#1-1},{#2+1}) rectangle ({#1+0},{#2+2});
    \draw[#5]        ({#1-4},{#2-1}) rectangle ({#1-3},{#2+0});
    \draw[#5]        ({#1+2},{#2-1}) rectangle ({#1+3},{#2+0});
    \draw[#5]        ({#1-1},{#2-2}) rectangle ({#1+0},{#2-1});
    \draw[#5]        ({#1-1},{#2+0}) rectangle ({#1+0},{#2+1});
}

\newcommand{\spotnord}[6]{%
    \draw[spot]  ({#1-1},{#2-1}) rectangle ({#1+0},{#2+0});
    \draw[#3]        ({#1-1},{#2-4}) rectangle ({#1+0},{#2-3});
    \draw[#4]        ({#1-1},{#2-3}) rectangle ({#1+0},{#2-2});
    \draw[#5]        ({#1-1},{#2-2}) rectangle ({#1+0},{#2-1});
    \draw            ({#1-.5},#2+2) node[spotlabel] {\Huge\bf #6};
}

\newcommand{\spotsued}[6]{%
    \draw[spot]  ({#1-1},{#2-1}) rectangle ({#1+0},{#2+0});
    \draw[#3]        ({#1-1},{#2+2}) rectangle ({#1+0},{#2+3});
    \draw[#4]        ({#1-1},{#2+1}) rectangle ({#1+0},{#2+2});
    \draw[#5]        ({#1-1},{#2+0}) rectangle ({#1+0},{#2+1});
    \draw            ({#1-.5},#2-3) node[spotlabel] {\Huge\bf #6};
}

\newcommand{\spotwest}[6]{%
    \draw[spot]  ({#1-1},{#2-1}) rectangle ({#1+0},{#2+0});
    \draw[#3]        ({#1+0},{#2-1}) rectangle ({#1+1},{#2+0});
    \draw[#4]        ({#1+1},{#2-1}) rectangle ({#1+2},{#2+0});
    \draw[#5]        ({#1+2},{#2-1}) rectangle ({#1+3},{#2+0});
    \draw            ({#1-3},#2-.5) node[spotlabel] {\Huge\bf #6};
}

\newcommand{\spotost}[6]{%
    \draw[spot]  ({#1-1},{#2-1}) rectangle ({#1+0},{#2+0});
    \draw[#3]        ({#1-2},{#2-1}) rectangle ({#1-1},{#2-0});
    \draw[#4]        ({#1-3},{#2-1}) rectangle ({#1-2},{#2+0});
    \draw[#5]        ({#1-4},{#2-1}) rectangle ({#1-3},{#2+0});
    \draw            ({#1+2},#2-.5) node[spotlabel] {\Huge\bf #6};
}

}

\newcommand{\calibspot}[4]{%
